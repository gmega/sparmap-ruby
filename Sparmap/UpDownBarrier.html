Overview

A simple barrier which admits both increments and decrements. Threads calling #wait will be blocked as long the barrier #count is greater than zero, and will be release when it becomes zero.

def count_down
  @mutex.synchronize do
    @count = [@count - 1, 0].max
    unpark_all if @count == 0
  end
end

def wait
  parking_spot = nil
  @mutex.synchronize do
    return if @count == 0
    parking_spot = Concurrent::Semaphore.new(0)
    @parked_threads << parking_spot
  end
  parking_spot.acquire
end
